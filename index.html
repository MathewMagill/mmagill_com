/* ============================ BLOB GAME LOGIC =============================== */ let score = 0; let level = 1; const promptsLevel1 = [ { chinese: "我喜欢苹果。", pinyin: "wǒ xǐhuān píngguǒ", likes: ["apple"] }, { chinese: "我不喜欢苹果。", pinyin: "wǒ bù xǐhuān píngguǒ", likes: ["banana"] }, { chinese: "我喜欢香蕉。", pinyin: "wǒ xǐhuān xiāngjiāo", likes: ["banana"] }, { chinese: "我不喜欢香蕉。", pinyin: "wǒ bù xǐhuān xiāngjiāo", likes: ["apple"] } ]; const promptsLevel2 = [ { chinese: "我喜欢草莓。", pinyin: "wǒ xǐhuān cǎoméi", likes: ["strawberry"] }, { chinese: "我不喜欢香蕉也不喜欢苹果。", pinyin: "wǒ bù xǐhuān xiāngjiāo yě bù xǐhuān píngguǒ", likes: ["strawberry"] } ]; const promptsLevel3 = [ { chinese: "我喜欢葡萄。", pinyin: "wǒ xǐhuān pútao", likes: ["grape"] }, { chinese: "我喜欢苹果也喜欢草莓。", pinyin: "wǒ xǐhuān píngguǒ yě xǐhuān cǎoméi", likes: ["apple", "strawberry"] }, { chinese: "我不喜欢香蕉也不喜欢苹果。", pinyin: "wǒ bù xǐhuān xiāngjiāo yě bù xǐhuān píngguǒ", likes: ["strawberry", "grape"] } ]; function randomPrompt() { if (level === 1) return promptsLevel1[Math.floor(Math.random() * promptsLevel1.length)]; if (level === 2) return promptsLevel2[Math.floor(Math.random() * promptsLevel2.length)]; return promptsLevel3[Math.floor(Math.random() * promptsLevel3.length)]; } let currentPrompt = randomPrompt(); document.getElementById("prompt").innerText = currentPrompt.chinese; document.getElementById("pinyin").innerText = currentPrompt.pinyin; const fruitEmoji = { apple: "🍎", banana: "🍌", strawberry: "🍓", grape: "🍇" }; let fruits = []; function spawnFruit(type = null) { let fruitType; if (type) { fruitType = type; } else { const pool = level === 1 ? ["apple", "banana"] : level === 2 ? ["apple", "banana", "strawberry"] : ["apple", "banana", "strawberry", "grape"]; fruitType = pool[Math.floor(Math.random() * pool.length)]; } fruits.push({ type: fruitType, x: Math.random() * 560 + 20, y: Math.random() * 360 + 20 }); } for (let i = 0; i < 5; i++) spawnFruit(); function ensureCorrectFruit() { const correctTypes = currentPrompt.likes; const exists = fruits.some(f => correctTypes.includes(f.type)); if (!exists) spawnFruit(correctTypes[Math.floor(Math.random() * correctTypes.length)]); } const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); let player = { x: 300, y: 200, size: 20, speed: 3 }; let keys = {}; function setupBlobTouchControls() { const touchControls = document.getElementById("touchControls"); if (!touchControls) return; const bindHold = (btnId, keyName) => { const btn = document.getElementById(btnId); if (!btn) return; const press = (e) => { e.preventDefault(); keys[keyName] = true; }; const release = (e) => { e.preventDefault(); keys[keyName] = false; }; // Pointer events work for both mouse + touch btn.addEventListener("pointerdown", press); btn.addEventListener("pointerup", release); btn.addEventListener("pointercancel", release); btn.addEventListener("pointerleave", release); // If the page loses focus, don't get "stuck" moving window.addEventListener("blur", () => (keys[keyName] = false)); }; bindHold("btnUp", "ArrowUp"); bindHold("btnDown", "ArrowDown"); bindHold("btnLeft", "ArrowLeft"); bindHold("btnRight", "ArrowRight"); } document.addEventListener("keydown", e => { if (["ArrowUp", "ArrowDown", "ArrowLeft", "ArrowRight"].includes(e.key)) { e.preventDefault(); } keys[e.key] = true; }); document.addEventListener("keyup", e => keys[e.key] = false); function updateBlobGame() { if (keys["w"] || keys["ArrowUp"]) player.y -= player.speed; if (keys["s"] || keys["ArrowDown"]) player.y += player.speed; if (keys["a"] || keys["ArrowLeft"]) player.x -= player.speed; if (keys["d"] || keys["ArrowRight"]) player.x += player.speed; player.x = Math.max(player.size, Math.min(canvas.width - player.size, player.x)); player.y = Math.max(player.size, Math.min(canvas.height - player.size, player.y)); fruits.forEach((fruit, i) => { const dx = fruit.x - player.x; const dy = fruit.y - player.y; const dist = Math.sqrt(dx * dx + dy * dy); if (dist < player.size) { if (currentPrompt.likes.includes(fruit.type)) { score++; player.size += 0.5; } else { score--; player.size = Math.max(10, player.size - 0.5); } fruits.splice(i, 1); spawnFruit(); if (score >= 10 && level === 1) level = 2; if (score >= 20 && level === 2) level = 3; currentPrompt = randomPrompt(); document.getElementById("prompt").innerText = currentPrompt.chinese; document.getElementById("pinyin").innerText = currentPrompt.pinyin; document.getElementById("score").innerText = "Score: " + score; ensureCorrectFruit(); } }); ensureCorrectFruit(); } function drawBlobGame() { ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.beginPath(); ctx.arc(player.x, player.y, player.size, 0, Math.PI * 2); ctx.fillStyle = "blue"; ctx.fill(); fruits.forEach(fruit => { ctx.font = "32px serif"; ctx.fillText(fruitEmoji[fruit.type], fruit.x - 10, fruit.y + 10); }); } function gameLoopBlob() { updateBlobGame(); drawBlobGame(); games.blobGameLoop = requestAnimationFrame(gameLoopBlob); } function startBlobGame() { score = 0; level = 1; player = { x: 300, y: 200, size: 20, speed: 3 }; fruits = []; for (let i = 0; i < 5; i++) spawnFruit(); currentPrompt = randomPrompt(); document.getElementById("prompt").innerText = currentPrompt.chinese; document.getElementById("pinyin").innerText = currentPrompt.pinyin; document.getElementById("score").innerText = "Score: 0"; setupBlobTouchControls(); gameLoopBlob(); }
